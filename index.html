    // === 可配置项 ===

    const WEDDING_DATE = new Date('2025-10-18T16:00:00+08:00'); // 修改为你的时间（含时区）


    // 倒计时逻辑

    function updateCountdown(){

      const now = new Date();

      let diff = Math.max(0, WEDDING_DATE - now);

      const d = Math.floor(diff / (1000*60*60*24)); diff -= d*24*60*60*1000;

      const h = Math.floor(diff / (1000*60*60)); diff -= h*60*60*1000;

      const m = Math.floor(diff / (1000*60)); diff -= m*60*1000;

      const s = Math.floor(diff / 1000);

      document.getElementById('dday').textContent = String(d).padStart(2,'0');

      document.getElementById('dh').textContent = String(h).padStart(2,'0');

      document.getElementById('dm').textContent = String(m).padStart(2,'0');

      document.getElementById('ds').textContent = String(s).padStart(2,'0');

    }

    setInterval(updateCountdown, 1000); updateCountdown();


    // RSVP 表单：示例为发送到邮箱（mailto）与本地提示

    function handleRSVP(e){

      e.preventDefault();

      const data = new FormData(e.target);

      const params = new URLSearchParams(data).toString();

      // TODO: 将下面地址改为你的邮箱或你的表单系统（Google Forms / Airtable / Netlify Forms）

      const mailto = `mailto:wedding@example.com?subject=${encodeURIComponent('婚礼回执 RSVP')}&body=${encodeURIComponent('\n姓名: '+data.get('name')+'\n电话: '+(data.get('phone')||'')+'\n是否出席: '+data.get('attend')+'\n同行人数: '+(data.get('guests')||'0')+'\n留言: '+(data.get('message')||''))}`;

      window.location.href = mailto;

      document.getElementById('thanks').style.display = 'block';

      e.target.reset();

    }
